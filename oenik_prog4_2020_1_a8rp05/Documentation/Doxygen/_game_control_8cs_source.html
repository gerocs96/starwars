98  private void GameControl_Loaded(object sender, RoutedEventArgs e)

100  Window win = Window.GetWindow(this);

102  if (win.FindResource("GCVM") is GameControlViewModel)
103  {
104  vm = win.FindResource("GCVM") as GameControlViewModel;
105  }
106  else

109  }
110 
111  if (vm.Model is null)
112  {
113  this.model = new GameModel(ships[vm.Ship], maps[vm.Map]);
114  }
115  else
116  {
117  this.model = new GameModel(vm.Model.Player, vm.Model.Map, vm.Model.Enemies, vm.Model.Bullets, vm.Model.Score, vm.Model.SpawnCounter, vm.Model.BasicSpawn, vm.Model.BossSpawn, vm.Model.SpeedIncrease);
118  }

163  private void MainTimer_Tick(object sender, EventArgs e)

173  AddScoreWindow asw = new AddScoreWindow(score);
174  Window win = Window.GetWindow(this);
175  asw.Top = win.Top + (win.Height / 2) - (asw.Width / 2);
176  asw.Left = win.Left + (win.Width / 2) - (asw.Height / 2);
177  asw.ShowDialog();
178  this.logic.AddScore(score);
179  this.StopGame();
180  }
